# Example Cron Configuration for EQ RNG Database Backups
# Copy these lines to your crontab using: crontab -e

# Daily backup at 2:00 AM, compressed, keep 30 days
0 2 * * * /path/to/eq_rng.com/utils/cron_backup.sh

# Docker backup every 6 hours, keep 28 backups (7 days worth)
0 */6 * * * /path/to/eq_rng.com/utils/backup_database.sh --docker --compress --keep 28

# Weekly backup on Sunday at 3:00 AM with extended retention
0 3 * * 0 /path/to/eq_rng.com/utils/backup_database.sh --docker --compress --keep 52 --output /backup/weekly

# Manual backup examples (run these directly from command line):

# Quick local backup
# /path/to/eq_rng.com/utils/backup_database.sh

# Production Docker backup with compression and cleanup
# /path/to/eq_rng.com/utils/backup_database.sh --docker --compress --keep 30

# Backup to custom location
# /path/to/eq_rng.com/utils/backup_database.sh --output /mnt/backup/database --compress

# Instructions:
# 1. Replace "/path/to/eq_rng.com" with your actual project path
# 2. Adjust timing and retention policies to your needs
# 3. Ensure the backup scripts are executable: chmod +x utils/*.sh
# 4. Test manually before adding to cron
# 5. Monitor backup logs in backups/backup.log (when using cron_backup.sh)
